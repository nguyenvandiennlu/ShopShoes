// chỉnh sửa lại trigger để tạo ID ngẫu nhiên, không sử dụng auto increment, tránh lỗi trùng id

DROP TRIGGER IF EXISTS before_insert_remember_token;
DROP TRIGGER IF EXISTS after_insert_remember_token;

DELIMITER ;;

CREATE TRIGGER before_insert_remember_token
BEFORE INSERT ON remember_token
FOR EACH ROW
BEGIN
    SET NEW.remember_token_id =
        CONCAT(
            'RT',
            UPPER(SUBSTRING(REPLACE(UUID(), '-', ''), 1, 8))
        );
END;;

DELIMITER ;

// xóa dữ liệu của để tránh bất đồng bộ dữ liệu
TRUNCATE TABLE remember_token;
